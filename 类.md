import UIKit
//类
class Person{
 var firstName:String
 var secondName:String
 //默认构造函数

 init(firstName:String,secondName:String) {
 self.firstName = firstName
 self.secondName = secondName
 }

 //自定义构造函数
 init?(nameString:String) {
 guard let separatorIndex = nameString.range(of: " ") else {
     return nil
 }

 guard let first = nameString.substring(to: separatorIndex.lowerBound) else {
     return nil
 }

 guard let second = nameString.substring(from: separatorIndex.lowerBound) else {
 return nil
 }

 self.firstName = first
 self.secondName = second

 func fullName()->String{
   print("the name is \(self.firstName) \(self.secondName)")
 }
 }
}

let person1 = Person(firstName:"liu",secondName:"binPeng")
let person2 = Person(nameString:"liu binPeng")
