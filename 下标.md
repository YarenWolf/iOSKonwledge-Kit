```
struct Vector3{
    var x:Double = 0.0
    var y:Double = 0.0
    var z:Double = 0.0
    // swift下标基础
    subscript(index:Int)->Double?{
        get{
            switch(index){
            case 0 :return x
            case 1: return  y
            case 2:return z
            default : return nil
            }
        }
        
        set{
            newValue
            guard let newValue = newValue else{
                return
            }
            switch(index){
            case 0: x = newValue
            case 1: y = newValue
            case 2: z = newValue
            default: return
            }
        }
    }
    
    subscript(axis:String)->Double?{
        switch(axis){
            case "x","X": return x
            case "y","Y": return y
            case "z","Z": return z
        default: return nil
        }
    }
}

var v = Vector3()

v.z

var v1 = Vector3(x: 3, y: 4, z: 5)

v1["Y"]
```

###多维下标


```
import UIKit

struct Matrix{
    var data:[[Double]]
    let r:Int
    let c:Int
    
    init(row:Int,coloumn:Int) {
        self.r = row
        self.c = coloumn
        data = [[Double]]()
        
        for _ in 0..<r {
            let aRow = [Array(repeatElement(0.0, count: coloumn))]
            data.append(contentsOf: aRow)
        }
    }
    
    subscript(x:Int,y:Int)->Double{
        get{
            //防止数组越界
            assert( x >= 0 && x < r && y >= 0 && y < c , "Index Out Of Range")
            return data[x][y]
        }
        
        set{
            //防止数组越界
            assert( x >= 0 && x < r && y >= 0 && y < c , "Index Out Of Range")
            data[x][y] = newValue
        }
    }
    
    
    subscript(row:Int)->[Double]{
        get{
            assert( row >= 0 && row < c , "Index Out Of Range")
             return data[row]
        }
        
        set(vector){
            assert( vector.count == c, "Column Number Does Not Match")
            data[row] = vector
        }
        
    }
}


var matrix = Matrix(row: 3, coloumn: 3)
matrix[2,2]

matrix[2,2] = 100
matrix[2,2]

matrix[2]
matrix[2][2]

matrix[2] = [1,2,3]
matrix[2,2]
matrix[2][2]
```
