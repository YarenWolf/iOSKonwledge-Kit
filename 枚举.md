###枚举
```
//枚举:用于标示一组值

enum GameEnding{
 case Win
 case Lose
 case Draw
}

var yourScore:Int = 100
var enemyScore:Int = 100
var reslut:GameEnding

if yourScore > enemyScore {
 reslut = GameEnding.Win
}else if yourScore < enemyScore{
 reslut = .Lose
}else{
 reslut = .Draw
}

switch reslut {
case .Win:
 print("you win")
 fallthrough
case .Lose:
 print("you loose")
 fallthrough
default:
 print("it is a draw")
}
```

###枚举关联默认值
```
//枚举关联默认值
//枚举:比c、c++更加灵活，不仅可以关联一组整形，还可以关联字符、等
enum Month:Int{
 case Jau = 1
 case Feb
 case Mar
}

var First:Month
First = Month.Jau
First.rawValue
var third:Month = Month.Mar
third.rawValue
```

###更佳灵活的枚举
```
//更加灵活的枚举使用方式
enum Barcode{
 case UPCA(Int,Int,Int,Int)
 case QRcode(String)
}

let productCodeA = Barcode.UPCA(2, 32, 243, 3)
let productCodeB:Barcode = .QRcode("this is a information")


switch productCodeB {
case .UPCA(let numberSystem , let manufuture , let shop , let kind):
 print("this product is UPCA and the value is (\(numberSystem),\(manufuture),\(shop),\(kind))")
case .QRcode(let information):
 print("this product is QRcode and the information is \(information)")
}
```
