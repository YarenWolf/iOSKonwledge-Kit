###枚举
```
//枚举:用于标示一组值

enum GameEnding{
 case Win
 case Lose
 case Draw
}

var yourScore:Int = 100
var enemyScore:Int = 100
var reslut:GameEnding

if yourScore > enemyScore {
 reslut = GameEnding.Win
}else if yourScore < enemyScore{
 reslut = .Lose
}else{
 reslut = .Draw
}

switch reslut {
case .Win:
 print("you win")
 fallthrough
case .Lose:
 print("you loose")
 fallthrough
default:
 print("it is a draw")
}
```
