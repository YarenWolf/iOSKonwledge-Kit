    func isPriziable()->Bool
}

    var name:String
    var score:Int
    var description: String{
        return name + "Socre:" + String(score)
    }
    func isPriziable() -> Bool {
        return score >= 60
    }

func ==(s1:Student,s2:Student)->Bool{
    return s1.score == s2.score
}
func <(s1:Student,s2:Student)->Bool{
    return s1.score < s2.score
}

    assert(seq.count > 0)
    return max(seq[0], seq[1])
