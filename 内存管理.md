import UIKit
class Person{
    var name:String
    init(name:String) {
        self.name = name
        print(name,"is coming")
    }
    func doSomething(){
        print(name,"is doing something")
    }
    
    deinit {
        print(name,"is leaving")
    }
}

var lbp:Person? = Person(name: "lbp")
lbp = nil
